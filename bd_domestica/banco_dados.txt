CREATE DATABASE bd_domestica;
USE bd_domestica;


CREATE TABLE perfil(
	id_perfil INT AUTO_INCREMENT,
	descricao_perfil VARCHAR(100),
	PRIMARY KEY(id_perfil)
)ENGINE INNODB;

CREATE TABLE usuario(
	id_usuario INT AUTO_INCREMENT,
	id_perfil INT,
	email VARCHAR(100),
	senha VARCHAR(100),
	data_matricula DATE,
	FOREIGN KEY(id_perfil)
	REFERENCES perfil(id_perfil),
	PRIMARY KEY(id_usuario)
)ENGINE INNODB;

CREATE TABLE cliente(
	id_clienteINT AUTO_INCREMENT,
	id_usuario INT,
	email VARCHAR(100),
	nome VARCHAR(100),
	rg VARCHAR(15),
	cpf VARCHAR(15),
	data_matricula DATE,
	FOREIGN KEY(id_usuario)
	REFERENCES usuario(id_usuario),
	PRIMARY KEY(id_cliente)
)ENGINE INNODB;

CREATE TABLE endereco(
	id_endereco INT AUTO_INCREMENT,
	id_usuario INT,
	id_cliente INT,
	cep VARCHAR(10)NOT NULL,
	rua VARCHAR(100),
	bairro VARCHAR(50),
	numero_residencia VARCHAR(10),
	cidade VARCHAR(50),
	estado CHAR(2),
	FOREIGN KEY(id_usuario)
	REFERENCES usuario(id_usuario),
	FOREIGN KEY(id_cliente)
	REFERENCES cliente(id_cliente),
	PRIMARY KEY(id_endereco)
	
)ENGINE INNODB;

CREATE TABLE servicos(
	id_servico INT AUTO_INCREMENT,
	data_servico DATE NOT NULL,
	horario_atendimento DATE,
	servico_feito CHAR(10),
	preco_servico VARCHAR(50),
	PRIMARY KEY(id_servico)
	
)ENGINE INNODB;


CREATE TABLE servicos_clientes(
	id_servico_cliente INT AUTO_INCREMENT,
	id_servico INT,
	id_cliente INT,
	atendente INT,
	id_endereco INT,
	data_servico_feito DATE,
	servico_feito CHAR(10),
	FOREIGN KEY(id_servico)
	REFERENCES servicos(id_servico),
	FOREIGN KEY (atendente)
	REFERENCES usuario(id_usuario),
	FOREIGN KEY(id_cliente)
	REFERENCES cliente(id_cliente),
	FOREIGN KEY(id_endereco)
	REFERENCES endereco(id_endereco),
	PRIMARY KEY(id_servico_cliente)
)ENGINE INNODB;

CREATE TABLE telefone(
	id_telefone INT AUTO_INCREMENT,
	id_cliente INT,
	id_usuario INT,
	tipo_telefone VARCHAR(20),
	numero_telefone VARCHAR(100),
	FOREIGN KEY(id_cliente)
	REFERENCES cliente(id_cliente),
	FOREIGN KEY(id_usuario)
	REFERENCES usuario(id_usuario),
	PRIMARY KEY(id_telefone)
)ENGINE INNODB;
